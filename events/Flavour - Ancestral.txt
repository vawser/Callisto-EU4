namespace = ixa_ancestral

# "Passing of Great Leader"
country_event = {
	id = ixa_ancestral.1
	title = ixa_ancestral.1.title
	desc = ixa_ancestral.1.desc
	picture = DRUIDCLAW_eventPicture
	
	trigger = {
		religion = shamanism
		ruler_religion = shamanism
		months_of_ruling = 180
		ruler_age = 55
		is_great_leader = yes
	}
	
	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = ixa_ancestral.1.option.a
		
		custom_tooltip = ixa_ancestral.1.option.a.tt
        
		hidden_effect = {
			exile_ruler_as = honored_great_person
			set_country_flag = honored_great_person_flag
            
			random = {
				chance = 50
				country_event = { id = ixa_ancestral.2 days = 3650 random = 1825 }
			}
		}
		tooltip = {
			kill_ruler = yes
		}
		add_years_of_income = -1
		add_prestige = 15
		random_list = {
			50 = {
				add_karma = -10
			}
			50 = {
				add_stability = 1
			}
		}
	}
    
	option = {
		name = ixa_ancestral.1.option.b
		
		custom_tooltip = ixa_ancestral.1.option.b.tt
        
		hidden_effect = {
			exile_ruler_as = disrespected_great_person
			set_country_flag = disrespected_great_person_flag
			random = {
				chance = 50
				
				country_event = { id = ixa_ancestral.3 days = 3650 random = 1825 }
			}
		}
		tooltip = {
			kill_ruler = yes
		}
		add_karma = 10
	}
    
	option = {
		name = ixa_ancestral.1.option.c
		
		custom_tooltip = ixa_ancestral.1.option.c.tt
        
		kill_ruler = yes
		add_years_of_income = -0.5
		add_prestige = 5
	}
}
