# Doubts of Ner'zhul
country_event = {
	id = ixa_rise_of_the_horde.3
	title = ixa_rise_of_the_horde.3.title
	desc = ixa_rise_of_the_horde.3.desc
	picture = ORCISH_INVASION_eventPicture
    
	trigger = {
        NOT = { has_global_flag = doubts_of_nerzhul }
        current_age = age_rise_of_the_horde
        had_global_flag = { flag = nerzhul_recruitment days = 365 }
		tag = SWM
	}
	
    mean_time_to_happen = {
		months = 24
	}
    
    immediate = {
        hidden_effect = {
            set_global_flag = doubts_of_nerzhul
            
            every_country = {
                limit = {
                    has_country_flag = set_side_for_nerzhul_event
                }
                if = { 
                    limit = { 
                        exists = SWM
                    }
                    remove_opinion = {
                        who = SWM
                        modifier = allies_of_nerzhul
                    }
                    SWM = {
                        remove_opinion = {
                            who = ROOT
                            modifier = allies_of_nerzhul
                        }
                    }
                }
                if = { 
                    limit = { 
                        exists = SWM
                    }
                    remove_opinion = {
                        who = SWM
                        modifier = enemies_of_nerzhul
                    }
                    SWM = {
                        remove_opinion = {
                            who = ROOT
                            modifier = enemies_of_nerzhul
                        }
                    }
                }
            }
        }
    }
    
	option = {
		name = ixa_rise_of_the_horde.3.option.a
		
        set_global_flag = guldan_recruitment
        set_country_flag = joined_guldan
        
        exile_ruler_as = exiled_nerzhul
        set_country_flag = exiled_nerzhul_flag
        
        define_ruler = {
            name = "Teron'gor"
            dynasty = "Shadowmoon"
            adm = 5
            dip = 4
            mil = 6
        }
        
        add_ruler_modifier = {
            name = kiljaedens_favour
            duration = -1
        }
    }
}
