namespace = ixa_setup_pirate

# Southsea Freebooters
country_event = {
	id = ixa_setup_pirate.1
	title = ixa_setup_pirate.1.title
	desc = ixa_setup_pirate.1.desc
	picture = BRAWL_eventPicture

    major = yes
	fire_only_once = yes
    
	trigger = {
        OR = {
            ai = no
            is_great_power = yes
        }
        NOT = { exists = SSP }
        NOT = { has_global_flag = setup_southsea_freebooters }
        current_age = age_scourge
    }
    
    immediate = {
        hidden_effect = {
            set_global_flag = setup_southsea_freebooters
        }
    }
    
    mean_time_to_happen = {
		months = 240
	}

	option = {
		name = ixa_setup_pirate.1.option.a
        
        area_tanaris_5 = {
            cede_province = SSP
            add_core = SSP
            change_culture = southsea_freebooter
            change_religion = avarice
        }
        
        ai_chance = {
            factor = 100
        }
	}
	option = {
		name = ixa_setup_pirate.1.option.b
        
        trigger = {
            area_tanaris_5 = {
                type = any
                owned_by = ROOT
            }
        }
        
        area_tanaris_5 = {
            change_culture = southsea_freebooter
            change_religion = avarice
        }
        
        ai_chance = {
            factor = 0
        }
	}
}
