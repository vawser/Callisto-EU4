country_event = {
	id = ixa_gnoll_flavour.2
	title = ixa_gnoll_flavour.2.title
	desc = ixa_gnoll_flavour.2.desc
	picture = GNOLL_eventPicture
	
	trigger = {
		culture_group = gnoll_group
        any_neighbor_country = {
            is_neighbor_of = ROOT
        }
	}
	
	mean_time_to_happen = {
		months = 360
	}

	immediate = {
		random_neighbor_country = {
			save_event_target_as = KidnapTarget
		}
	}
	
	option = {
		name = ixa_gnoll_flavour.2.opt1
		
		reverse_add_opinion = {
			who = event_target:KidnapTarget
			modifier = dishonorable_act
		}
		reverse_add_casus_belli = {
			target = event_target:KidnapTarget
			type = cb_conquest
			months = 12
		}
		add_country_modifier = { 
			name = fresh_meat
			duration = 365 
			desc = fresh_meat.desc
			hidden = no
		}
	}
	
	option = {
		name = ixa_gnoll_flavour.2.opt2
		
		reverse_add_opinion = {
			who = event_target:KidnapTarget
			modifier = ate_our_citizen
		}
		reverse_add_casus_belli = {
			target = event_target:KidnapTarget
			type = cb_conquest
			months = 12
		}
		add_prestige = 15
	}
	
	option = {
		name = ixa_gnoll_flavour.2.opt3
		
		add_prestige = -10
	}
}
